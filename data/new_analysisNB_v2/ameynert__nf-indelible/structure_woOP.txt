digraph  {

aggregate;
score;
blast;
annotate;
denovo;
annotated_filter;
denovo_filter;
in;
out;

denovo -> denovo_filter  [key=0];

denovo_filter -> out  [key=0];
in -> fetch  [key=0];
}
